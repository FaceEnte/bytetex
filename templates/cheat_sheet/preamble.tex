%! Custom subtitle command
\newcommand{\subtitle}[1]{%
  {\Large #1}
}

%! Custom author info block
\ExplSyntaxOn
\NewDocumentCommand{\authorinfo}{m m}{%
  \vfill
  \begin{flushleft}
    \fontsize{11pt}{13pt}\selectfont
    \begin{tabular}{@{} l @{\quad} l}
      \textbf{\mytitledatelabel} \hspace{2pt} & #1 \\         %! Date
      \clist_map_inline:nn {#2} {%
        \textbf{\mytitleauthorlabel} \hspace{2pt} & ##1 \\    %! Authors
      }
    \end{tabular}
  \end{flushleft}
}
\ExplSyntaxOff

%! Custom titlepage command
\newcommand{\customtitlepage}[4]{%
  \begin{titlepage}
    \centering
    \vspace*{120pt}
    \LARGE #1 \par                                    %! Title
    \vspace{0.5cm}
    \Large \subtitle{#2} \par                         %! Subtitle
    \vfill
    \authorinfo{#3}{#4}                               %! Date and Authors
  \end{titlepage}
  \newpage
}
